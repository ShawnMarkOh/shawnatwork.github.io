// This function deals a random card from the deck. function dealCard() { var card = Math.floor(Math.random() * 52) + 1; return card; } // This function gets the image URL for a card. function getCardImage(card) { var suit = card % 13; var value = Math.floor(card / 13) + 1; var imageUrl = "card-" + suit + "-" + value + ".png"; return imageUrl; } // This function updates the UI with the new cards. function updateUI(player1Card, player2Card) { document.getElementById("player1").innerHTML = ""; document.getElementById("player2").innerHTML = ""; } // This function determines the winner of the round. function getWinner(player1Card, player2Card) { if (player1Card > player2Card) { return "Player 1"; } else if (player1Card < player2Card) { return "Player 2"; } else { return "Tie"; } } // This function runs the game. function runGame() { var player1Card = dealCard(); var player2Card = dealCard(); updateUI(player1Card, player2Card); var winner = getWinner(player1Card, player2Card); document.getElementById("winner").innerHTML = winner; } // This function is called when the page loads. window.onload = runGame;
